rr library(data.table) getmr <- function(files){ outdf <- data.frame() for(i in 1:length(files)){ r <- fread(files[i], data.table=FALSE, header=T) sub <- subset(r, Chr == 1 & Pos > 270512569 & Pos < 270513049) sub\(totcov <- sub\)count_methy + sub\(count_unmethy tem <- data.frame(f=files[i], mr = mean(sub\)methy_per), cov=mean(sub$totcov)) outdf <- rbind(outdf, tem) } return(outdf) }

files <- list.files(path=/common/jyanglab/shared/Gen_Xu/methylation_level/Landrace, pattern=, full.names = TRUE) out <- getmr(files)
